109 {
110 constexpr auto const kChallengeMBCI = 0xdeadbeef;
111
112 host->MMIOTest = kChallengeMBCI;
113
114 if (host->MMIOTest == kChallengeMBCI)
115 {
116 return (host->Esb[kMBCIESBSz - 1] << 16) | (host->Esb[kMBCIESBSz - 2] << 8) | (host->Esb[kMBCIESBSz - 3] & 0xFF);
117 }
118
119 return kChallengeMBCI;
120 }
