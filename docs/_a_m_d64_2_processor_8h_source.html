173inline Bool hal_has_msr() noexcept {
174 static UInt32 eax, unused, edx; // eax, edx
175
176 __get_cpuid(1, &eax, &unused, &unused, &edx);
177
178 // edx returns the flag for MSR (which is 1 shifted to 5.)
179 return edx & (1 << 5);
180}

190inline UInt32 hal_get_msr(UInt32 msr, UInt32* lo, UInt32* hi) noexcept {
191 if (!lo || !hi) return 0;
192
193 asm volatile("rdmsr" : "=a"(*lo), "=d"(*hi) : "c"(msr));
194
195 return *lo + *hi;
196}
