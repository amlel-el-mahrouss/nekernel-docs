193 {
194 static UInt32 eax, unused, edx; // eax, edx
195
196 __get_cpuid(1, &eax, &unused, &unused, &edx);

210 inline UInt32 hal_get_msr(UInt32 msr, UInt32* lo, UInt32* hi) noexcept
211 {
212 if (!lo || !hi)
213 return 0;
214
215 asm volatile("rdmsr"
216 : "=a"(*lo), "=d"(*hi)
217 : "c"(msr));
218
219 return *lo + *hi;
220 }
