199 inline Bool hal_has_msr() noexcept
200 {
201 static UInt32 eax, unused, edx; // eax, edx
202
203 __get_cpuid(1, &eax, &unused, &unused, &edx);

205 // edx returns the flag for MSR (which is 1 shifted to 5.)

217 inline UInt32 hal_get_msr(UInt32 msr, UInt32* lo, UInt32* hi) noexcept
218 {
219 if (!lo || !hi)
220 return 0;
221
222 asm volatile("rdmsr"
223 : "=a"(*lo), "=d"(*hi)
224 : "c"(msr));
225
226 return *lo + *hi;
227 }
