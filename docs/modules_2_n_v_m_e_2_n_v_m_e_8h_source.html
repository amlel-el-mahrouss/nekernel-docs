82 {
83 if (entry == nullptr)
84 return false;
85
86 if (baseAddress == 0)
87 return false;
88
89 entry->fOpcode = Opcode;
90
91 entry->CommandSpecific[5] = (baseAddress & 0xFF);
92 entry->CommandSpecific[6] = static_cast<UInt32>(baseAddress);
93
94 if (!provideIdentify)
95 {
96 entry->CommandSpecific[9] = identLoAndQueueSizeHi;
97 entry->CommandSpecific[10] = flagsLoAndQueueComplIdHi;
98 }
99 else
100 {
101 entry->CommandSpecific[9] = identify;
102
103 if (namespaceIdentify)
104 {
105 entry->CommandSpecific[0] = 1;
106 }
107 }
108
109 return true;
