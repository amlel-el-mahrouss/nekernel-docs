76 struct JsonStreamReader final
77 {
78 STATIC Json In(const Char* full_array)
79 {
80 auto start_val = '{';
81 auto end_val = '}';
82 Boolean probe_value = false;
83
84 if (full_array[0] != start_val)
85 {
86 if (full_array[0] != '[')
87 return Json::kNull;
88
89 start_val = '[';
90 end_val = ']';
91
92 probe_value = true;
93 }
94
95 SizeT len = rt_string_len(full_array);
96
97 SizeT key_len = 0;
98 SizeT value_len = 0;
99
100 Json type(kMaxJsonPath, kMaxJsonPath);
101
102 for (SizeT i = 1; i < len; ++i)
103 {
104 if (full_array[i] == '\r' ||
105 full_array[i] == '\n')
106 continue;
107
108 if (probe_value)
109 {
110 if (full_array[i] == end_val ||
111 full_array[i] == ',')
112 {
113 probe_value = false;
114
115 ++value_len;
116 }
117 else
118 {
119 type.AsValue().Data()[value_len] = full_array[i];
120
121 ++value_len;
122 }
123 }
124 else
125 {
126 if (start_val == '[')
127 continue;
128
129 if (full_array[i] == ':')
130 {
131 probe_value = true;
132 type.AsKey().Data()[key_len] = 0;
133 ++key_len;
134 }
135 else
136 {
137 type.AsKey().Data()[key_len] = full_array[i];
138
139 ++key_len;
140 }
141 }
142 }
143
144 type.AsValue().Data()[value_len] = 0;
145
146 return type;
147 }
148 };
