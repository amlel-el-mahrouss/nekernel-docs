191Boolean CompilerFrontendCPlusPlus::Compile(std::string text, std::string file) {
192 if (text.empty()) return false;
193
194 std::size_t index = 0UL;
195 std::vector<std::pair<LibCompiler::CompilerKeyword, std::size_t>> keywords_list;
196
197 for (auto& keyword : kKeywords) {
198 if (text.find(keyword.keyword_name) != std::string::npos) {
199 switch (keyword.keyword_kind) {
200 case LibCompiler::kKeywordKindCommentInline: {
201 break;
202 }
203 default:
204 break;
205 }
206
207 if (text[text.find(keyword.keyword_name) - 1] == '+' &&
208 keyword.keyword_kind == LibCompiler::KeywordKind::kKeywordKindVariableAssign)
209 continue;
210
211 if (text[text.find(keyword.keyword_name) - 1] == '-' &&
212 keyword.keyword_kind == LibCompiler::KeywordKind::kKeywordKindVariableAssign)
213 continue;
214
215 if (text[text.find(keyword.keyword_name) + 1] == '=' &&
216 keyword.keyword_kind == LibCompiler::KeywordKind::kKeywordKindVariableAssign)
217 continue;
218
219 keywords_list.emplace_back(std::make_pair(keyword, index));
220 ++index;
221 }
222 }
223
224 LibCompiler::SyntaxLeafList::SyntaxLeaf syntax_tree;
225
226 for (auto& keyword : keywords_list) {
227 switch (keyword.first.keyword_kind) {
228 case LibCompiler::KeywordKind::kKeywordKindClass: {
229 ++kOnClassScope;
230 break;
231 }
232 case LibCompiler::KeywordKind::kKeywordKindIf: {
233 auto expr = text.substr(
234 text.find(keyword.first.keyword_name) + keyword.first.keyword_name.size() + 1,
235 text.find(")") - 1);
236
237 if (expr.find(">=") != std::string::npos) {
238 auto left = text.substr(
239 text.find(keyword.first.keyword_name) + keyword.first.keyword_name.size() + 2,
240 expr.find("<=") + strlen("<="));
241 auto right = text.substr(expr.find(">=") + strlen(">="), text.find(")") - 1);
242
243 size_t i = right.size() - 1;
244
245 if (i < 1) break;
246
247 try {
248 while (!std::isalnum(right[i])) {
249 right.erase(i, 1);
250 --i;
251 }
252
253 right.erase(0, i);
254 } catch (...) {
255 right.erase(0, i);
256 }
257
258 i = left.size() - 1;
259 try {
260 while (!std::isalnum(left[i])) {
261 left.erase(i, 1);
262 --i;
263 }
264
265 left.erase(0, i);
266 } catch (...) {
267 left.erase(0, i);
268 }
269
270 if (!isdigit(left[0]) || !isdigit(right[0])) {
271 auto indexRight = 0UL;
272
273 auto& valueOfVar = !isdigit(left[0]) ? left : right;
274
275 for (auto pairRight : kRegisterMap) {
276 ++indexRight;
277
278 if (pairRight != valueOfVar) {
279 auto& valueOfVarOpposite = isdigit(left[0]) ? left : right;
280
281 syntax_tree.fUserValue +=
282 "mov " + kRegisterList[indexRight + 1] + ", " + valueOfVarOpposite + "\n";
283 syntax_tree.fUserValue += "cmp " + kRegisterList[kRegisterMap.size() - 1] + "," +
284 kRegisterList[indexRight + 1] + "\n";
285
286 goto done_iterarting_on_if;
287 }
288
289 auto& valueOfVarOpposite = isdigit(left[0]) ? left : right;
290
291 syntax_tree.fUserValue +=
292 "mov " + kRegisterList[indexRight + 1] + ", " + valueOfVarOpposite + "\n";
293 syntax_tree.fUserValue += "cmp " + kRegisterList[kRegisterMap.size() - 1] + ", " +
294 kRegisterList[indexRight + 1] + "\n";
295
296 break;
297 }
298 }
299
300 done_iterarting_on_if:
301
302 std::string fnName = text;
303 fnName.erase(fnName.find(keyword.first.keyword_name));
304
305 for (auto& ch : fnName) {
306 if (ch == ' ') ch = '_';
307 }
308
309 syntax_tree.fUserValue +=
310 "jge __OFFSET_ON_TRUE_LC\nsegment .code64 __OFFSET_ON_TRUE_LC:\n";
311 }
312
313 break;
314 }
315 case LibCompiler::KeywordKind::kKeywordKindFunctionStart: {
316 for (auto& ch : text) {
317 if (isdigit(ch)) {
318 goto dont_accept;
319 }
320 }
321
322 goto accept;
323
324 dont_accept:
325 return false;
326
327 accept:
328 std::string fnName = text;
329 size_t indexFnName = 0;
330
331 // this one is for the type.
332 for (auto& ch : text) {
333 ++indexFnName;
334
335 if (ch == '\t') break;
336 if (ch == ' ') break;
337 }
338
339 fnName = text.substr(indexFnName);
340
341 if (text.find("return ") != std::string::npos) {
342 text.erase(0, text.find("return "));
343 break;
344 }
345
346 if (text.ends_with(";") && text.find("return") == std::string::npos)
347 goto lc_write_assembly;
348 else if (text.size() <= indexFnName)
349 Detail::print_error("Invalid function name: " + fnName, file);
350
351 indexFnName = 0;
352
353 for (auto& ch : fnName) {
354 if (ch == ' ' || ch == '\t') {
355 if (fnName[indexFnName - 1] != ')')
356 Detail::print_error("Invalid function name: " + fnName, file);
357 }
358
359 ++indexFnName;
360 }
361
362 if (fnName.find("(") != LibCompiler::String::npos) {
363 fnName.erase(fnName.find("("));
364 }
365
366 syntax_tree.fUserValue = "public_segment .code64 __LIBCOMPILER_" + fnName + "\n";
367 ++kFunctionEmbedLevel;
368
369 kOriginMap.push_back({"__LIBCOMPILER_" + fnName, kOrigin});
370
371 break;
372
373 lc_write_assembly:
374 auto it = std::find_if(kOriginMap.begin(), kOriginMap.end(),
375 [&fnName](std::pair<std::string, std::uintptr_t> pair) -> bool {
376 return fnName == pair.first;
377 });
378
379 if (it != kOriginMap.end()) {
380 std::stringstream ss;
381 ss << std::hex << it->second;
382
383 syntax_tree.fUserValue = "jmp " + ss.str() + "\n";
384 kOrigin += 1UL;
385 }
386 }
387 case LibCompiler::KeywordKind::kKeywordKindFunctionEnd: {
388 if (kOnClassScope) --kOnClassScope;
389
390 if (text.ends_with(";")) break;
391
392 --kFunctionEmbedLevel;
393
394 if (kRegisterMap.size() > kRegisterList.size()) {
395 --kFunctionEmbedLevel;
396 }
397
398 if (kFunctionEmbedLevel < 1) kRegisterMap.clear();
399
400 break;
401 }
402 case LibCompiler::KeywordKind::kKeywordKindEndInstr:
403 case LibCompiler::KeywordKind::kKeywordKindVariableInc:
404 case LibCompiler::KeywordKind::kKeywordKindVariableDec:
405 case LibCompiler::KeywordKind::kKeywordKindVariableAssign: {
406 std::string valueOfVar = "";
407
408 if (keyword.first.keyword_kind == LibCompiler::KeywordKind::kKeywordKindVariableInc) {
409 valueOfVar = text.substr(text.find("+=") + 2);
410 } else if (keyword.first.keyword_kind ==
411 LibCompiler::KeywordKind::kKeywordKindVariableDec) {
412 valueOfVar = text.substr(text.find("-=") + 2);
413 } else if (keyword.first.keyword_kind ==
414 LibCompiler::KeywordKind::kKeywordKindVariableAssign) {
415 valueOfVar = text.substr(text.find("=") + 1);
416 } else if (keyword.first.keyword_kind == LibCompiler::KeywordKind::kKeywordKindEndInstr) {
417 break;
418 }
419
420 while (valueOfVar.find(";") != std::string::npos &&
421 keyword.first.keyword_kind != LibCompiler::KeywordKind::kKeywordKindEndInstr) {
422 valueOfVar.erase(valueOfVar.find(";"));
423 }
424
425 std::string varName = text;
426
427 if (keyword.first.keyword_kind == LibCompiler::KeywordKind::kKeywordKindVariableInc) {
428 varName.erase(varName.find("+="));
429 } else if (keyword.first.keyword_kind ==
430 LibCompiler::KeywordKind::kKeywordKindVariableDec) {
431 varName.erase(varName.find("-="));
432 } else if (keyword.first.keyword_kind ==
433 LibCompiler::KeywordKind::kKeywordKindVariableAssign) {
434 varName.erase(varName.find("="));
435 } else if (keyword.first.keyword_kind == LibCompiler::KeywordKind::kKeywordKindEndInstr) {
436 varName.erase(varName.find(";"));
437 }
438
439 static Boolean typeFound = false;
440
441 for (auto& keyword : kKeywords) {
442 if (keyword.keyword_kind == LibCompiler::kKeywordKindType) {
443 if (text.find(keyword.keyword_name) != std::string::npos) {
444 if (text[text.find(keyword.keyword_name)] == ' ') {
445 typeFound = false;
446 continue;
447 }
448
449 typeFound = true;
450 }
451 }
452 }
453
454 std::string instr = "mov ";
455
456 if (typeFound &&
457 keyword.first.keyword_kind != LibCompiler::KeywordKind::kKeywordKindVariableInc &&
458 keyword.first.keyword_kind != LibCompiler::KeywordKind::kKeywordKindVariableDec) {
459 if (kRegisterMap.size() > kRegisterList.size()) {
460 ++kFunctionEmbedLevel;
461 }
462
463 while (varName.find(" ") != std::string::npos) {
464 varName.erase(varName.find(" "), 1);
465 }
466
467 while (varName.find("\t") != std::string::npos) {
468 varName.erase(varName.find("\t"), 1);
469 }
470
471 for (size_t i = 0; !isalnum(valueOfVar[i]); i++) {
472 if (i > valueOfVar.size()) break;
473
474 valueOfVar.erase(i, 1);
475 }
476
477 constexpr auto kTrueVal = "true";
478 constexpr auto kFalseVal = "false";
479
480 if (valueOfVar == kTrueVal) {
481 valueOfVar = "1";
482 } else if (valueOfVar == kFalseVal) {
483 valueOfVar = "0";
484 }
485
486 std::size_t indexRight = 0UL;
487
488 for (auto pairRight : kRegisterMap) {
489 ++indexRight;
490
491 if (pairRight != valueOfVar) {
492 if (valueOfVar[0] == '\"') {
493 syntax_tree.fUserValue = "segment .data64 __LIBCOMPILER_LOCAL_VAR_" + varName +
494 ": db " + valueOfVar + ", 0\n\n";
495 syntax_tree.fUserValue += instr + kRegisterList[kRegisterMap.size() - 1] + ", " +
496 "__LIBCOMPILER_LOCAL_VAR_" + varName + "\n";
497 kOrigin += 1UL;
498 } else {
499 syntax_tree.fUserValue =
500 instr + kRegisterList[kRegisterMap.size() - 1] + ", " + valueOfVar + "\n";
501 kOrigin += 1UL;
502 }
503
504 goto done;
505 }
506 }
507
508 if (((int) indexRight - 1) < 0) {
509 if (valueOfVar[0] == '\"') {
510 syntax_tree.fUserValue = "segment .data64 __LIBCOMPILER_LOCAL_VAR_" + varName +
511 ": db " + valueOfVar + ", 0\n";
512 syntax_tree.fUserValue += instr + kRegisterList[kRegisterMap.size()] + ", " +
513 "__LIBCOMPILER_LOCAL_VAR_" + varName + "\n";
514 kOrigin += 1UL;
515 } else {
516 syntax_tree.fUserValue =
517 instr + kRegisterList[kRegisterMap.size()] + ", " + valueOfVar + "\n";
518 kOrigin += 1UL;
519 }
520
521 goto done;
522 }
523
524 if (valueOfVar[0] != '\"' && valueOfVar[0] != '\'' && !isdigit(valueOfVar[0])) {
525 for (auto pair : kRegisterMap) {
526 if (pair == valueOfVar) goto done;
527 }
528
529 Detail::print_error("Variable not declared: " + varName, file);
530 return false;
531 }
532
533 done:
534 for (auto& keyword : kKeywords) {
535 if (keyword.keyword_kind == LibCompiler::kKeywordKindType &&
536 varName.find(keyword.keyword_name) != std::string::npos) {
537 varName.erase(varName.find(keyword.keyword_name), keyword.keyword_name.size());
538 break;
539 }
540 }
541
542 kRegisterMap.push_back(varName);
543
544 break;
545 }
546
547 if (kKeywords[keyword.second - 1].keyword_kind == LibCompiler::kKeywordKindType ||
548 kKeywords[keyword.second - 1].keyword_kind == LibCompiler::kKeywordKindTypePtr) {
549 syntax_tree.fUserValue = "\n";
550 continue;
551 }
552
553 if (keyword.first.keyword_kind == LibCompiler::KeywordKind::kKeywordKindEndInstr) {
554 syntax_tree.fUserValue = "\n";
555 continue;
556 }
557
558 if (keyword.first.keyword_kind == LibCompiler::KeywordKind::kKeywordKindVariableInc) {
559 instr = "add ";
560 } else if (keyword.first.keyword_kind ==
561 LibCompiler::KeywordKind::kKeywordKindVariableDec) {
562 instr = "sub ";
563 }
564
565 std::string varErrCpy = varName;
566
567 while (varName.find(" ") != std::string::npos) {
568 varName.erase(varName.find(" "), 1);
569 }
570
571 while (varName.find("\t") != std::string::npos) {
572 varName.erase(varName.find("\t"), 1);
573 }
574
575 std::size_t indxReg = 0UL;
576
577 for (size_t i = 0; !isalnum(valueOfVar[i]); i++) {
578 if (i > valueOfVar.size()) break;
579
580 valueOfVar.erase(i, 1);
581 }
582
583 while (valueOfVar.find(" ") != std::string::npos) {
584 valueOfVar.erase(valueOfVar.find(" "), 1);
585 }
586
587 while (valueOfVar.find("\t") != std::string::npos) {
588 valueOfVar.erase(valueOfVar.find("\t"), 1);
589 }
590
591 constexpr auto kTrueVal = "true";
592 constexpr auto kFalseVal = "false";
593
595
596 if (valueOfVar == kTrueVal) {
597 valueOfVar = "1";
598 } else if (valueOfVar == kFalseVal) {
599 valueOfVar = "0";
600 }
601
602 for (auto pair : kRegisterMap) {
603 ++indxReg;
604
605 if (pair != varName) continue;
606
607 std::size_t indexRight = 0ul;
608
609 for (auto pairRight : kRegisterMap) {
610 ++indexRight;
611
612 if (pairRight != varName) {
613 syntax_tree.fUserValue =
614 instr + kRegisterList[kRegisterMap.size()] + ", " + valueOfVar + "\n";
615 kOrigin += 1UL;
616 continue;
617 }
618
619 syntax_tree.fUserValue =
620 instr + kRegisterList[indexRight - 1] + ", " + valueOfVar + "\n";
621 kOrigin += 1UL;
622 break;
623 }
624
625 break;
626 }
627
628 if (syntax_tree.fUserValue.empty()) {
629 Detail::print_error("Variable not declared: " + varName, file);
630 }
631
632 break;
633 }
634 case LibCompiler::KeywordKind::kKeywordKindReturn: {
635 try {
636 auto pos = text.find("return") + strlen("return") + 1;
637 std::string subText = text.substr(pos);
638 subText = subText.erase(subText.find(";"));
639 size_t indxReg = 0UL;
640
641 if (subText[0] != '\"' && subText[0] != '\'') {
642 if (!isdigit(subText[0])) {
643 for (auto pair : kRegisterMap) {
644 ++indxReg;
645
646 if (pair != subText) continue;
647
648 syntax_tree.fUserValue = "mov rax, " + kRegisterList[indxReg - 1] + "\nret\n";
649 kOrigin += 1UL;
650
651 break;
652 }
653 } else {
654 syntax_tree.fUserValue = "mov rax, " + subText + "\nret\n";
655 kOrigin += 1UL;
656
657 break;
658 }
659 } else {
660 syntax_tree.fUserValue = "__LIBCOMPILER_LOCAL_RETURN_STRING: db " + subText +
661 ", 0\nmov rcx, __LIBCOMPILER_LOCAL_RETURN_STRING\n";
662 syntax_tree.fUserValue += "mov rax, rcx\nret\n";
663 kOrigin += 1UL;
664
665 break;
666 }
667
668 if (syntax_tree.fUserValue.empty()) {
669 if (subText.find("(") != std::string::npos) {
670 subText.erase(subText.find("("));
671
672 auto it =
673 std::find_if(kOriginMap.begin(), kOriginMap.end(),
674 [&subText](std::pair<std::string, std::uintptr_t> pair) -> bool {
675 return pair.first.find(subText) != std::string::npos;
676 });
677
678 if (it == kOriginMap.end())
679 Detail::print_error("Invalid return value: " + subText, file);
680
681 std::stringstream ss;
682 ss << it->second;
683
684 syntax_tree.fUserValue = "jmp " + ss.str() + "\nret\n";
685 kOrigin += 1UL;
686 break;
687 }
688 }
689
690 break;
691 } catch (...) {
692 syntax_tree.fUserValue = "ret\n";
693 kOrigin += 1UL;
694 }
695 }
696 default: {
697 continue;
698 }
699 }
700 }
701
702 kState.fOutputValue = syntax_tree.fUserValue;
703 return true;
704}

714class AssemblyCPlusPlusInterface final ASSEMBLY_INTERFACE {
715 public:
716 explicit AssemblyCPlusPlusInterface() = default;
717 ~AssemblyCPlusPlusInterface() override = default;
718
719 LIBCOMPILER_COPY_DEFAULT(AssemblyCPlusPlusInterface);
720
721 [[maybe_unused]] Int32 Arch() noexcept override {
722 return LibCompiler::AssemblyFactory::kArchAMD64;
723 }
724
725 Int32 CompileToFormat(std::string src, Int32 arch) override {
726 if (kCompilerFrontend == nullptr) return kExitNO;
727
728 /* @brief copy contents wihtout extension */
729 std::ifstream src_fp = std::ifstream(src, std::ios::in);
730
731 std::string dest = src;
732 dest += ".masm";
733
734 std::string line_source;
735
736 std::ofstream out(dest);
737
738 while (std::getline(src_fp, line_source)) {
739 kCompilerFrontend->Compile(line_source, src);
740 out << kState.fOutputValue;
741 }
742
743 out.flush();
744
745 if (kAcceptableErrors > 0) return kExitNO;
746
747 return kExitOK;
748 }
749};
