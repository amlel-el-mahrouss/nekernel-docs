344 class NeFileSystemJournal final
345 {
346 private:
347 NEFS_CATALOG_STRUCT* mNode{nullptr};
348
349 public:
350 explicit NeFileSystemJournal(const char* stamp = nullptr)
351 {
352 if (!stamp)
353 {
354 kout << "Invalid: Journal Stamp, using default name.\r";
355 return;
356 }
357
358 (void)(kout << "Info: Journal stamp: " << stamp << kendl);
359 rt_copy_memory((VoidPtr)stamp, this->mStamp, rt_string_len(stamp));
360 }
361
362 ~NeFileSystemJournal() = default;
363
364 NE_COPY_DEFAULT(NeFileSystemJournal)
365
366 Bool CreateJournal(NeFileSystemParser* parser)
367 {
368 if (!parser)
369 return NO;
370
371 delete parser->CreateCatalog("/etc/xml/", 0, kNeFSCatalogKindDir);
372 mNode = parser->CreateCatalog(mStamp);
373
374 if (!mNode)
375 return NO;
376
377 return YES;
378 }
379
380 Bool GetJournal(NeFileSystemParser* parser)
381 {
382 if (!parser)
383 return NO;
384
385 auto node = parser->GetCatalog(mStamp);
386
387 if (node)
388 {
389 mNode = node;
390 return YES;
391 }
392
393 return NO;
394 }
395
396 Bool ReleaseJournal()
397 {
398 if (mNode)
399 {
400 delete mNode;
401 mNode = nullptr;
402 return YES;
403 }
404
405 return NO;
406 }
407
408 Bool CommitJournal(NeFileSystemParser* parser,
409 Char* xml_data,
410 Char* journal_name)
411 {
412 if (!parser ||
413 !mNode)
414 return NO;
415
416 NEFS_FORK_STRUCT new_fork{};
417
418 rt_copy_memory(mStamp, new_fork.CatalogName, rt_string_len(mStamp));
419 rt_copy_memory(journal_name, new_fork.ForkName, rt_string_len(journal_name));
420
421 new_fork.ResourceKind = 0;
422 new_fork.ResourceId = 0;
423 new_fork.ResourceFlags = 0;
424 new_fork.DataSize = rt_string_len(xml_data);
425 new_fork.Kind = kNeFSRsrcForkKind;
426
427 if (!parser->CreateFork(new_fork))
428 return NO;
429
430 (void)(kout << "XML Commited: " << xml_data << "\r\nTo Journal Fork: " << journal_name << kendl);
431
432 auto ret = parser->WriteCatalog(new_fork.CatalogName, YES, xml_data, rt_string_len(xml_data), new_fork.ForkName);
433
434 return ret;
435 }
436
437 private:
438 Char mStamp[kNeFSCatalogNameLen] = {"/etc/xml/journal" kNeFSJournalExt};
439 };
