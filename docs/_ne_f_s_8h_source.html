349 class NeFileSystemJournal final
350 {
351 private:
352 NEFS_CATALOG_STRUCT* mNode{nullptr};
353
354 public:
355 explicit NeFileSystemJournal(const char* stamp = nullptr)
356 {
357 if (!stamp)
358 {
359 kout << "Invalid: Journal stamp, using default name.\r";
360 return;
361 }
362
363 (Void)(kout << "Info: Journal stamp: " << stamp << kendl);
364 rt_copy_memory((VoidPtr)stamp, this->mStamp, rt_string_len(stamp));
365 }
366
367 ~NeFileSystemJournal() = default;
368
369 NE_COPY_DEFAULT(NeFileSystemJournal)
370
371 Bool CreateJournal(NeFileSystemParser* parser)
372 {
373 if (!parser)
374 return NO;
375
376 delete parser->CreateCatalog("/etc/xml/", 0, kNeFSCatalogKindDir);
377 mNode = parser->CreateCatalog(mStamp);
378
379 if (!mNode)
380 return NO;
381
382 return YES;
383 }
384
385 Bool GetJournal(NeFileSystemParser* parser)
386 {
387 if (!parser)
388 return NO;
389
390 auto node = parser->GetCatalog(mStamp);
391
392 if (node)
393 {
394 mNode = node;
395 return YES;
396 }
397
398 return NO;
399 }
400
401 Bool ReleaseJournal()
402 {
403 if (mNode)
404 {
405 delete mNode;
406 mNode = nullptr;
407 return YES;
408 }
409
410 return NO;
411 }
412
413 Bool CommitJournal(NeFileSystemParser* parser,
414 Char* xml_data,
415 Char* journal_name)
416 {
417 if (!parser ||
418 !mNode)
419 return NO;
420
421 NEFS_FORK_STRUCT new_fork{};
422
423 rt_copy_memory(mStamp, new_fork.CatalogName, rt_string_len(mStamp));
424 rt_copy_memory(journal_name, new_fork.ForkName, rt_string_len(journal_name));
425
426 new_fork.ResourceKind = 0;
427 new_fork.ResourceId = 0;
428 new_fork.ResourceFlags = 0;
429 new_fork.DataSize = rt_string_len(xml_data);
430 new_fork.Kind = kNeFSRsrcForkKind;
431
432 if (!parser->CreateFork(new_fork))
433 return NO;
434
435 (void)(kout << "XML commit: " << xml_data << " to fork: " << journal_name << kendl);
436
437 auto ret = parser->WriteCatalog(new_fork.CatalogName, YES, xml_data, rt_string_len(xml_data), new_fork.ForkName);
438
439 return ret;
440 }
441
442 private:
443 Char mStamp[kNeFSCatalogNameLen] = {"/etc/xml/journal" kNeFSJournalExt};
444 };
