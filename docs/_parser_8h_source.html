112inline bool find_word(std::string haystack, std::string needle) noexcept {
113 auto index = haystack.find(needle);
114
115 // check for needle validity.
116 if (index == std::string::npos) return false;
117
118 // declare lambda
119 auto not_part_of_word = [&](int index) {
120 if (std::isspace(haystack[index]) || std::ispunct(haystack[index])) return true;
121
122 if (index <= 0 || index >= haystack.size()) return true;
123
124 return false;
125 };
126
127 return not_part_of_word(index - 1) && not_part_of_word(index + needle.size());
128}
129
134inline std::size_t find_word_range(std::string haystack, std::string needle) noexcept {
135 auto index = haystack.find(needle);
136
137 // check for needle validity.
138 if (index == std::string::npos) return false;
139
140 if (!isalnum((haystack[index + needle.size() + 1])) &&
141 !isdigit(haystack[index + needle.size() + 1]) &&
142 !isalnum((haystack[index - needle.size() - 1])) &&
143 !isdigit(haystack[index - needle.size() - 1])) {
144 return index;
145 }
146
147 return false;
148}
