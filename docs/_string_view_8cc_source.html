85const char* StringBuilder::FromInt(const char* fmt, int i) {
86 if (!fmt) return ("-1");
87
88 auto ret_len = 8 + string_length(fmt);
89 char* ret = new char[ret_len];
90
91 if (!ret) return ("-1");
92
93 memset(ret, 0, ret_len);
94
95 CharType result[sizeof(int64_t)];
96
97 if (!to_str(result, sizeof(int64_t), i)) {
98 delete[] ret;
99 return ("-1");
100 }
101
102 const auto fmt_len = string_length(fmt);
103 const auto res_len = string_length(result);
104
105 for (SizeType idx = 0; idx < fmt_len; ++idx) {
106 if (fmt[idx] == '%') {
107 SizeType result_cnt = idx;
108
109 for (auto y_idx = 0; y_idx < res_len; ++y_idx) {
110 ret[y_idx] = result[result_cnt];
111 ++result_cnt;
112 }
113
114 break;
115 }
116
117 ret[idx] = fmt[idx];
118 }
119
120 return ret; /* Copy that ret into a buffer, Alloca allocates to the stack */
121}
