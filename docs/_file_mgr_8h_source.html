209 ErrorOr<Int64> Write(SizeT offset, const VoidPtr data, SizeT len) noexcept
210 {
211 if (this->fFileRestrict != kFileMgrRestrictReadWrite &&
212 this->fFileRestrict != kFileMgrRestrictReadWriteBinary &&
213 this->fFileRestrict != kFileMgrRestrictWrite &&
214 this->fFileRestrict != kFileMgrRestrictWriteBinary)
215 return ErrorOr<Int64>(kErrorInvalidData);
216
217 if (data == nullptr)
218 return ErrorOr<Int64>(kErrorInvalidData);
219
220 auto man = FSClass::GetMounted();
221
222 if (man)
223 {
224 man->Write(offset, fFile, data, len);
225 return ErrorOr<Int64>(0);
226 }
227
228 return ErrorOr<Int64>(kErrorInvalidData);
229 }
230
231 ErrorOr<Int64> Write(const Char* name, const VoidPtr data, SizeT len) noexcept
232 {
233 if (this->fFileRestrict != kFileMgrRestrictReadWrite &&
234 this->fFileRestrict != kFileMgrRestrictReadWriteBinary &&
235 this->fFileRestrict != kFileMgrRestrictWrite &&
236 this->fFileRestrict != kFileMgrRestrictWriteBinary)
237 return ErrorOr<Int64>(kErrorInvalidData);
238
239 if (data == nullptr)
240 return ErrorOr<Int64>(kErrorInvalidData);
241
242 auto man = FSClass::GetMounted();
243
244 if (man)
245 {
246 man->Write(name, fFile, data, 0, len);
247 return ErrorOr<Int64>(0);
248 }
249
250 return ErrorOr<Int64>(kErrorInvalidData);
251 }
252
253 VoidPtr Read(const Char* name, SizeT sz) noexcept
254 {
255 if (this->fFileRestrict != kFileMgrRestrictReadWrite &&
256 this->fFileRestrict != kFileMgrRestrictReadWriteBinary &&
257 this->fFileRestrict != kFileMgrRestrictRead &&
258 this->fFileRestrict != kFileMgrRestrictReadBinary)
259 return nullptr;
260
261 NE_UNUSED(sz);
262
263 auto man = FSClass::GetMounted();
264
265 if (man)
266 {
267 VoidPtr ret = man->Read(name, fFile, kFileReadAll, 0);
268 return ret;
269 }
270
271 return nullptr;
272 }
273
274 VoidPtr Read(SizeT offset, SizeT sz)
275 {
276 if (this->fFileRestrict != kFileMgrRestrictReadWrite &&
277 this->fFileRestrict != kFileMgrRestrictReadWriteBinary &&
278 this->fFileRestrict != kFileMgrRestrictRead &&
279 this->fFileRestrict != kFileMgrRestrictReadBinary)
280 return nullptr;
281
282 auto man = FSClass::GetMounted();
283
284 if (man)
285 {
286 man->Seek(fFile, offset);

288
289 return ret;
290 }
291
292 return nullptr;

341 template <typename Encoding, typename Class>
342 inline FileStream<Encoding, Class>::FileStream(const Encoding* path,
343 const Encoding* restrict_type)
344 : fFile(Class::GetMounted()->Open(path, restrict_type))
345 {
346 SizeT kRestrictCount = kRestrictMax;
347 const FileRestrictKind kRestrictList[] = {
348 {
349 .fRestrict = kRestrictR,
350 .fMappedTo = kFileMgrRestrictRead,
351 },
352 {
353 .fRestrict = kRestrictRB,
354 .fMappedTo = kFileMgrRestrictReadBinary,
355 },
356 {
357 .fRestrict = kRestrictRWB,
358 .fMappedTo = kFileMgrRestrictReadWriteBinary,
359 },
360 {
361 .fRestrict = kRestrictW,
362 .fMappedTo = kFileMgrRestrictWrite,
363 },
364 {
365 .fRestrict = kRestrictWB,
366 .fMappedTo = kFileMgrRestrictReadWrite,
367 }};
368
369 for (SizeT index = 0; index < kRestrictCount; ++index)
370 {
371 if (rt_string_cmp(restrict_type, kRestrictList[index].fRestrict,
372 rt_string_len(kRestrictList[index].fRestrict)) == 0)
373 {
374 fFileRestrict = kRestrictList[index].fMappedTo;
375 break;
376 }
377 }
378
379 kout << "new file: " << path << ".\r";
380 }
