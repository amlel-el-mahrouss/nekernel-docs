213 ErrorOr<Int64> Write(const SizeT offset, const VoidPtr data, SizeT len) noexcept
214 {
215 if (this->fFileRestrict != kFileMgrRestrictReadWrite &&
216 this->fFileRestrict != kFileMgrRestrictReadWriteBinary &&
217 this->fFileRestrict != kFileMgrRestrictWrite &&
218 this->fFileRestrict != kFileMgrRestrictWriteBinary)
219 return ErrorOr<Int64>(kErrorInvalidData);
220
221 if (data == nullptr)
222 return ErrorOr<Int64>(kErrorInvalidData);
223
224 auto man = FSClass::GetMounted();
225
226 if (man)
227 {
228 man->Write(offset, fFile, data, len);
229 return ErrorOr<Int64>(0);
230 }
231
232 return ErrorOr<Int64>(kErrorInvalidData);
233 }

235 ErrorOr<Int64> Write(const Char* name, const VoidPtr data, SizeT len) noexcept
236 {
237 if (this->fFileRestrict != kFileMgrRestrictReadWrite &&
238 this->fFileRestrict != kFileMgrRestrictReadWriteBinary &&
239 this->fFileRestrict != kFileMgrRestrictWrite &&
240 this->fFileRestrict != kFileMgrRestrictWriteBinary)
241 return ErrorOr<Int64>(kErrorInvalidData);
242
243 if (data == nullptr)
244 return ErrorOr<Int64>(kErrorInvalidData);
245
246 auto man = FSClass::GetMounted();
247
248 if (man)
249 {
250 man->Write(name, fFile, data, 0, len);
251 return ErrorOr<Int64>(0);
252 }
253
254 return ErrorOr<Int64>(kErrorInvalidData);
255 }
256
257 VoidPtr Read(const Char* name, const SizeT sz) noexcept
258 {
259 if (this->fFileRestrict != kFileMgrRestrictReadWrite &&
260 this->fFileRestrict != kFileMgrRestrictReadWriteBinary &&
261 this->fFileRestrict != kFileMgrRestrictRead &&
262 this->fFileRestrict != kFileMgrRestrictReadBinary)
263 return nullptr;
264
265 auto man = FSClass::GetMounted();
266
267 if (man)
268 {
269 VoidPtr ret = man->Read(name, fFile, kFileReadAll, 0);
270 return ret;
271 }
272
273 return nullptr;

343 template <typename Encoding, typename Class>
344 inline FileStream<Encoding, Class>::FileStream(const Encoding* path,
345 const Encoding* restrict_type)
346 : fFile(Class::GetMounted()->Open(path, restrict_type))
347 {
348 const SizeT kRestrictCount = kRestrictMax;
349 const FileRestrictKind kRestrictList[] = {
350 {
351 .fRestrict = kRestrictR,
352 .fMappedTo = kFileMgrRestrictRead,
353 },
354 {
355 .fRestrict = kRestrictRB,
356 .fMappedTo = kFileMgrRestrictReadBinary,
357 },
358 {
359 .fRestrict = kRestrictRWB,
360 .fMappedTo = kFileMgrRestrictReadWriteBinary,
361 },
362 {
363 .fRestrict = kRestrictW,
364 .fMappedTo = kFileMgrRestrictWrite,
365 },
366 {
367 .fRestrict = kRestrictWB,
368 .fMappedTo = kFileMgrRestrictReadWrite,
369 }};
370
371 for (SizeT index = 0; index < kRestrictCount; ++index)
372 {
373 if (rt_string_cmp(restrict_type, kRestrictList[index].fRestrict,
374 rt_string_len(kRestrictList[index].fRestrict)) == 0)
375 {
376 fFileRestrict = kRestrictList[index].fMappedTo;
377 break;
378 }
379 }
380
381 kout << "new file: " << path << ".\r";
382 }
