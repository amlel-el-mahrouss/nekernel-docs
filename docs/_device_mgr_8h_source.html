48 IDeviceObject(const IDeviceObject<T>&) = default;

50 public:
51 virtual IDeviceObject<T>& operator<<(T Data) {
52 fOut(this, Data);

54 }
55
56 virtual IDeviceObject<T>& operator>>(T Data) {
57 fIn(this, Data);
58 return *this;
