21 inline uint8_t gf_mul(uint8_t a, uint8_t b)
22 {
23 uint8_t res = 0;
24
25 for (; b; b >>= 1)
26 {
27 if (b & 1)
28 res ^= a;
29 if (a & 0x80)
30 a = (a << 1) ^ min_poly;
31 else
32 a <<= 1;
33 }
34
35 return res;
36 }

198 // Reed-Solomon Ecc generator polynomial for the given degree.
199 constexpr void gf_gen_poly(int degree, uint8_t* poly)
200 {
201 SetMem(poly, 0, degree);
202
203 uint8_t root = poly[degree - 1] = 1;
204
205 for (int i = 0; i < degree; ++i)
206 {
207 for (int j = 0; j < degree - 1; ++j)
208 poly[j] = gf_mul(poly[j], root) ^ poly[j + 1];
209 poly[degree - 1] = gf_mul(poly[degree - 1], root);
210 root = (root << 1) ^ ((root >> 7) * 0x11d);
211 }
212 }
213
214 // Polynomial division if Galois Field.
215 constexpr void gf_poly_div(uint8_t* dividend, size_t len, uint8_t* divisor, int degree, uint8_t* result)
216 {
217 SetMem(result, 0, degree);
218
219 for (size_t i = 0; i < len; ++i)
220 {
221 uint8_t factor = dividend[i] ^ result[0];
222 MoveMem(&result[0], &result[1], degree - 1);
223 result[degree - 1] = 0;
224 for (int j = 0; j < degree; ++j)
225 result[j] ^= gf_mul(divisor[j], factor);
226 }
227 }

416 template <int V>
417 bool Qr<V>::encode_data(const char* data, size_t len, Ecc ecc, uint8_t* out)
418 {
419 Mode mode = select_mode(data, len);
420
421 size_t n_bits =
422 (N_DAT_CAPACITY - ECC_CODEWORDS_PER_BLOCK[ecc][V] * N_ECC_BLOCKS[ecc][V])
423 << 3;
424 size_t pos = 0;
425
426 add_bits(1 << mode, 4, out, pos);
427 add_bits(len, cci(V, mode), out, pos);
428
429 if (mode == M_NUMERIC)
430 {
431 const size_t triplets = len / 3;
432 const size_t triplets_size = triplets * 3;
433 const size_t rem = len % 3;
434 const size_t rem_bits = rem == 2 ? 7 : rem == 1 ? 4
435 : 0;
436 const size_t total_size = 10 * triplets + rem_bits;
437
438 if (pos + total_size > n_bits)
439 return false;
440
441 char buf[4] = {};
442
443 for (size_t i = 0; i < triplets_size; i += 3)
444 {
445 buf[0] = data[i];
446 buf[1] = data[i + 1];
447 buf[2] = data[i + 2];
448
449 uint16_t num = StringToLong(buf, NULL, 10);
450 add_bits(num, 10, out, pos);
451 }
452
453 if (rem)
454 {
455 buf[0] = data[triplets_size];
456 buf[1] = data[triplets_size + 1];
457 buf[rem] = 0;
458
459 uint16_t num = StringToLong(buf, NULL, 10);
460 add_bits(num, rem_bits, out, pos);
461 }
462 }
463 else if (mode == M_ALPHANUMERIC)
464 {
465 if (pos + 11 * (int(len & ~1ul) / 2) > n_bits)
466 return false;
467
468 for (int i = 0; i < int(len & ~1ul); i += 2)
469 {
470 uint16_t num = alphanumeric(data[i]) * 45 + alphanumeric(data[i + 1]);
471 add_bits(num, 11, out, pos);
472 }
473 if (len & 1)
474 {
475 if (pos + 6 > n_bits)
476 return false;
477
478 add_bits(alphanumeric(data[len - 1]), 6, out, pos);
479 }
480 }
481 else if (mode == M_BYTE)
482 {
483 if (pos + len * 8 > n_bits)
484 return false;
485
486 for (size_t i = 0; i < len; ++i)
487 add_bits(data[i], 8, out, pos);
488 }
489 else
490 {
491 if (pos + 13 * (len / 2) > n_bits)
492 return false;
493
494 for (size_t i = 0; i < len; i += 2)
495 {
496 uint16_t val = ((uint8_t)data[i]) | (((uint8_t)data[i + 1]) << 8);
497 uint16_t res = 0;
498 val -= val < 0x9FFC ? 0x8140 : 0xC140;
499 res += val & 0xff;
500 res += (val >> 8) * 0xc0;
501 add_bits(res, 13, out, pos);
502 }
503 }
504
505 size_t padding = n_bits - pos;
506 size_t i = 0;
507
508 add_bits(0, padding > 4 ? 4 : padding, out, pos);
509
510 if (pos & 7)
511 add_bits(0, (8 - pos) & 7, out, pos);
512
513 while (pos < n_bits)
514 add_bits(++i & 1 ? 0xec : 0x11, 8, out, pos);
515
516 return true;
517 }

519 template <int V>
520 void Qr<V>::encode_ecc(uint8_t* data, Ecc ecc, uint8_t* out)
521 {
522 int n_blocks = N_ECC_BLOCKS[ecc][V];
523 int ecc_len = ECC_CODEWORDS_PER_BLOCK[ecc][V];
524
525 int n_data_bytes = N_DAT_CAPACITY - ecc_len * n_blocks;
526
527 int n_short_blocks = n_blocks - N_DAT_CAPACITY % n_blocks;
528 int short_len = N_DAT_CAPACITY / n_blocks - ecc_len;
529
530 uint8_t gen_poly[30];
531 uint8_t ecc_buf[30];
532
533 gf_gen_poly(ecc_len, gen_poly);
534
535 uint8_t* data_ptr = data;
536
537 for (int i = 0; i < n_blocks; ++i)
538 {
539 int data_len = short_len;
540
541 if (i >= n_short_blocks)
542 ++data_len;
543
544 gf_poly_div(data_ptr, data_len, gen_poly, ecc_len, ecc_buf);
545
546 for (int j = 0, k = i; j < data_len; ++j, k += n_blocks)
547 {
548 if (j == short_len)
549 k -= n_short_blocks;
550 out[k] = data_ptr[j];
551 }
552 for (int j = 0, k = n_data_bytes + i; j < ecc_len; ++j, k += n_blocks)
553 out[k] = ecc_buf[j];
554
555 data_ptr += data_len;
556 }
557 }

560 void Qr<V>::add_data(uint8_t* data, uint8_t* patterns)
561 {
562 int data_pos = 0;
563
564 for (int x = SIDE - 1; x >= 1; x -= 2)
565 {
566 if (x == 6)
567 x = 5;
568
569 for (int i = 0; i < SIDE; ++i)
570 {

574 if (!get_arr_bit(patterns, coord))
575 {
576 if (get_bit_r(data, data_pos))
577 set_arr_bit(code, coord);
578
579 ++data_pos;
580 }
581
582 if (!get_arr_bit(patterns, coord - 1))
583 {
584 if (get_bit_r(data, data_pos))
585 set_arr_bit(code, coord - 1);

587 ++data_pos;

630 {

633
634 uint32_t rem = V;
635
636 for (uint8_t i = 0; i < 12; ++i)

638

640
641 for (int x = 0; x < 6; ++x)
642 {

658 template <int V>
659 void Qr<V>::add_format(Ecc ecc, int mask)
660 {
661 int data = (ecc ^ 1) << 3 | mask;
662 int rem = data;
663
664 for (int i = 0; i < 10; i++)
665 rem = (rem << 1) ^ ((rem >> 9) * 0b10100110111);
666
667 int res = (data << 10 | rem) ^ 0b101010000010010;
668
669 for (int i = 0; i < 6; ++i)
670 {
671 if ((res >> i) & 1)
672 {
673 set_arr_bit(code, SIDE * 8 + SIDE - 1 - i);
674 set_arr_bit(code, SIDE * i + 8);
675 }
676 else
677 {
678 clr_arr_bit(code, SIDE * 8 + SIDE - 1 - i);
679 clr_arr_bit(code, SIDE * i + 8);
680 }
681 }
682
683 for (int i = 6; i < 8; ++i)
684 {
685 if ((res >> i) & 1)
686 {
687 set_arr_bit(code, SIDE * 8 + SIDE - 1 - i);
688 set_arr_bit(code, SIDE * (i + 1) + 8);
689 }
690 else
691 {
692 clr_arr_bit(code, SIDE * 8 + SIDE - 1 - i);
693 clr_arr_bit(code, SIDE * (i + 1) + 8);
694 }
695 }
696
697 if ((res >> 8) & 1)
698 {
699 set_arr_bit(code, SIDE * 8 + 7);
700 set_arr_bit(code, SIDE * (SIDE - 7) + 8);
701 }
702 else
703 {
704 clr_arr_bit(code, SIDE * 8 + 7);
705 clr_arr_bit(code, SIDE * (SIDE - 7) + 8);
706 }
707
708 for (int i = 9, j = 5; i < 15; ++i, --j)
709 {
710 if ((res >> i) & 1)

800 template <int V>
801 template <bool H>
802 int Qr<V>::rule_1_3_score()
803 {
804 constexpr int y_max = H ? N_BITS : SIDE;
805 constexpr int x_max = H ? SIDE : N_BITS;
806 constexpr int y_step = H ? SIDE : 1;
807 constexpr int x_step = H ? 1 : SIDE;
808
809 int res = 0;
810
811 for (int y = 0; y < y_max; y += y_step)
812 {
813 bool color = get_arr_bit(code, y);
814 int finder = color;
815 int cnt = 1;
816 for (int x = 1; x < x_max; x += x_step)
817 {
818 if (get_arr_bit(code, y + x) == color)
819 {
820 ++cnt;
821 if (cnt == 5)
822 res += 3;
823 if (cnt > 5)
824 ++res;
825 }
826 else
827 {
828 color = !color;
829 cnt = 1;
830 }
831 // Finder-like
832 finder = ((finder << 1) & 0x7ff) | color;
833 if (x >= x_step * 10)
834 {
835 if (finder == 0x05d || finder == 0x5d0)
836 res += 40;
837 }
838 }
839 }
840 return res;
841 }

843 template <int V>
844 int Qr<V>::penalty_score()
845 {
846 int res = 0;
847
848 res += rule_1_3_score<true>();
849 res += rule_1_3_score<false>();
850
851 for (int y = 0; y < N_BITS - SIDE; y += SIDE)
852 {
853 for (int x = 0; x < SIDE - 1; ++x)
854 {
855 bool c = get_arr_bit(code, y + x);
856
857 if (c == get_arr_bit(code, y + x + 1) &&
858 c == get_arr_bit(code, y + x + SIDE) &&
859 c == get_arr_bit(code, y + x + SIDE + 1))
860 res += 3;
861 }
862 }
863
864 int black = 0;
865 for (int y = 0; y < N_BITS; y += SIDE)
866 {
867 for (int x = 0; x < SIDE; ++x)
868 black += get_arr_bit(code, y + x);
869 }
870 res += abs((black * 100) / N_BITS - 50) / 5 * 10;
871
872 return res;
873 }

876 int Qr<V>::select_mask(Ecc ecc, uint8_t* patterns)
877 {
878 unsigned min_score = -1;
879 unsigned score = 0;
880 uint8_t mask = 0;
881

883 {
884 add_format(ecc, i);
885 apply_mask(i, patterns);
886 score = penalty_score();

888 {
889 mask = i;
890 min_score = score;
891 }
892 apply_mask(i, patterns);
893 }

897 template <int V>
898 void Qr<V>::apply_mask(int mask, uint8_t* patterns)
899 {
900 for (int y = 0, dy = 0; y < SIDE; ++y, dy += SIDE)
901 {
902 for (int x = 0; x < SIDE; ++x)
903 {
904 int coord = dy + x;
905
906 if (get_arr_bit(patterns, coord))
907 continue;
908
909 bool keep = true;
910
911 switch (mask)
912 {
913 case 0:
914 keep = (x + y) & 1;
915 break;
916 case 1:
917 keep = y & 1;
918 break;
919 case 2:
920 keep = x % 3;
921 break;
922 case 3:
923 keep = (x + y) % 3;
924 break;
925 case 4:
926 keep = (y / 2 + x / 3) & 1;
927 break;
928 case 5:
929 keep = x * y % 2 + x * y % 3;
930 break;
931 case 6:
932 keep = (x * y % 2 + x * y % 3) & 1;
933 break;
934 case 7:
935 keep = ((x + y) % 2 + x * y % 3) & 1;
936 break;
937 }
938
939 if (!keep)
940 {
941 if (get_arr_bit(code, coord))
942 clr_arr_bit(code, coord);
943 else
944 set_arr_bit(code, coord);
945 }
946 }
947 }
