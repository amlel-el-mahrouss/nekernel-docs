229 Bool IsPartitionValid() noexcept
230 {
231 fDiskDev.Leak().mBase = (kNeFSRootCatalogStartAddress);
232 fDiskDev.Leak().mSize = BootDev::kSectorSize;
233
234 Char buf[BootDev::kSectorSize] = {0};
235
236 fDiskDev.Read(buf, BootDev::kSectorSize);
237
238 NEFS_ROOT_PARTITION_BLOCK* blockPart = reinterpret_cast<NEFS_ROOT_PARTITION_BLOCK*>(buf);
239
240 BootTextWriter writer;
241
242 for (SizeT indexMag = 0UL; indexMag < kNeFSIdentLen; ++indexMag)
243 {
244 if (blockPart->Ident[indexMag] != kNeFSIdent[indexMag])
245 return false;
246 }
247
248 if (blockPart->DiskSize != this->fDiskDev.GetDiskSize() ||
249 blockPart->DiskSize < 1 ||
250 blockPart->SectorSize != BootDev::kSectorSize ||
251 blockPart->Version != kNeFSVersionInteger ||
252 blockPart->StartCatalog == 0)
253 {
254 return false;
255 }
256 else if (blockPart->PartitionName[0] == 0)
257 {
258 return false;
259 }
260
261 writer.Write(L"BootZ: Partition: ").Write(blockPart->PartitionName).Write(L" is healthy.\r");
262
263 return true;
264 }
265
266 private:
271 Boolean WriteCatalogList(BFileDescriptor* blob, SizeT blob_sz, NEFS_ROOT_PARTITION_BLOCK& part)
272 {
273 if (blob_sz < sizeof(NEFS_CATALOG_STRUCT))
274 return NO;
275
276 if (!blob)
277 return NO;
278
279 Lba startLba = part.StartCatalog;
280 BootTextWriter writer;
281
282 NEFS_CATALOG_STRUCT catalogKind{0};
283
284 constexpr auto cNeFSCatalogPadding = 4;
285
286 catalogKind.PrevSibling = startLba;
287 catalogKind.NextSibling = (startLba + sizeof(NEFS_CATALOG_STRUCT) * cNeFSCatalogPadding);
288
290 catalogKind.Kind = blob->fKind;
291 catalogKind.Flags |= kNeFSFlagCreated;
292 catalogKind.CatalogFlags = kNeFSStatusUnlocked;
293
294 --part.FreeCatalog;
295 --part.FreeSectors;
296
297 CopyMem(catalogKind.Name, blob->fFileName, StrLen(blob->fFileName));
298
299 fDiskDev.Leak().mBase = startLba;
300 fDiskDev.Leak().mSize = sizeof(NEFS_CATALOG_STRUCT);
301
302 fDiskDev.Write((Char*)&catalogKind, sizeof(NEFS_CATALOG_STRUCT));
303
304 writer.Write(L"BootZ: Wrote directory: ").Write(blob->fFileName).Write(L"\r");
305
306 return true;
307 }

338 Boot::ThrowError(L"Drive-Too-Tiny", L"Can't format a NeFS partition here.");

352 part.SectorCount = fDiskDev.GetSectorsCount();
353 part.FreeSectors = fDiskDev.GetSectorsCount();
354 part.StartCatalog = kNeFSCatalogStartAddress;
355 part.DiskSize = fDiskDev.GetDiskSize();
356 part.Flags = kNeFSPartitionTypeBoot | kNeFSPartitionTypeStandard;
357
358 fDiskDev.Leak().mBase = kNeFSRootCatalogStartAddress;
359 fDiskDev.Leak().mSize = sizeof(NEFS_ROOT_PARTITION_BLOCK);
