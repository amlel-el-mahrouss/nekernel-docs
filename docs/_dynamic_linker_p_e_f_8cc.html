620 .find(kLdDefineSymbol) != LibCompiler::String::npos &&
621 LibCompiler::String(command_headers[sub_command_header_index].Name).find(kLdDynamicSym) ==
622 LibCompiler::String::npos) {

625 }

628 continue;
629 }
630

642 }

644 }

649 }
650
651 return LIBCOMPILER_EXEC_ERROR;
652 }
653
654 // step 2.5: write program bytes.
655
656 for (auto& struct_of_blob : kObjectBytes) {
657 output_fc.write(struct_of_blob.mBlob.data(), struct_of_blob.mBlob.size());
658 }
659
660 if (kVerbose) {
661 kOutCon << "Wrote contents of: " << kOutput << "\n";
662 }
663
664 // step 3: check if we have those symbols
665
666 std::vector<LibCompiler::String> unreferenced_symbols;
667
668 for (auto& command_hdr : command_headers) {
669 if (auto it =
670 std::find(not_found.begin(), not_found.end(), LibCompiler::String(command_hdr.Name));
671 it != not_found.end()) {
672 unreferenced_symbols.emplace_back(command_hdr.Name);
673 }
674 }
675
676 if (!unreferenced_symbols.empty()) {
677 for (auto& unreferenced_symbol : unreferenced_symbols) {
678 kOutCon << "Undefined symbol " << unreferenced_symbol << "\n";
679 }
680
681 return LIBCOMPILER_EXEC_ERROR;
682 }
683
684 if ((!kStartFound || kDuplicateSymbols) &&
685 (std::filesystem::exists(kOutput) || !unreferenced_symbols.empty())) {
686 if (kVerbose) {
687 kOutCon << "File: " << kOutput << ", is corrupt, removing file...\n";
688 }
689
690 return LIBCOMPILER_EXEC_ERROR;
691 }
692
693 return LIBCOMPILER_SUCCESS;
